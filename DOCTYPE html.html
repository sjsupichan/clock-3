        const apiKey = ""; // Provided by environment

        const markersContainer = document.getElementById('markers-container');

        for (let i = 0; i < 12; i++) {

        }

 

        function updateClock() {

            const now = new Date();

            const seconds = now.getSeconds();

            const minutes = now.getMinutes();

            const hours = now.getHours();

 

            const secDeg = (seconds / 60) * 360;

            const minDeg = (minutes / 60) * 360 + (seconds / 60) * 6;

            const hourDeg = (hours / 12) * 360 + (minutes / 60) * 30;

 

            secHand.style.transform = `translateX(-50%) rotate(${secDeg}deg)`;

            minHand.style.transform = `translateX(-50%) rotate(${minDeg}deg)`;

            hourHand.style.transform = `translateX(-50%) rotate(${hourDeg}deg)`;

 

            const h = hours.toString().padStart(2, '0');

            const m = minutes.toString().padStart(2, '0');

            const s = seconds.toString().padStart(2, '0');

            digitalDisplay.textContent = `${h}:${m}:${s}`;

 

            const options = { weekday: 'long', month: 'long', day: 'numeric' };

            dateDisplay.textContent = now.toLocaleDateString('en-US', options);

        }

        setInterval(updateClock, 1000);

        updateClock();

        function showTab(tabName) {

            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

            document.getElementById(`${tabName}-tab`).classList.add('active');

 

            const btns = ['clock', 'stopwatch', 'ai'];

            btns.forEach(b => {

                const btn = document.getElementById(`${b}-tab-btn`);

                if (b === tabName) {

                    btn.className = "flex-1 py-2 rounded-xl text-sm font-medium transition-all bg-sky-500 text-white";

                } else {

                    btn.className = "flex-1 py-2 rounded-xl text-sm font-medium transition-all text-slate-400 hover:text-white";

                }

            });

        }

 

        // --- STOPWATCH LOGIC ---

        let swStartTime, swElapsedTime = 0, swInterval, isRunning = false, laps = [];

        const swDisplay = document.getElementById('sw-display');

        const swStartBtn = document.getElementById('sw-start-btn');

        const swLapBtn = document.getElementById('sw-lap-btn');

        const lapsContainer = document.getElementById('laps-container');

 

        function formatTime(ms) {

            const minutes = Math.floor(ms / 60000);

            const seconds = Math.floor((ms % 60000) / 1000);

            const centiseconds = Math.floor((ms % 1000) / 10);

            return `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}.${centiseconds.toString().padStart(2, '0')}`;

        }

 

        function toggleStopwatch() {

            if (!isRunning) {

                isRunning = true;

                swStartTime = Date.now() - swElapsedTime;

                swInterval = setInterval(() => {

                    swElapsedTime = Date.now() - swStartTime;

                    swDisplay.textContent = formatTime(swElapsedTime);

                }, 10);

                swStartBtn.textContent = 'STOP';

                swStartBtn.classList.replace('bg-sky-500', 'bg-orange-500');

                swLapBtn.disabled = false;

            } else {

                isRunning = false;

                clearInterval(swInterval);

                swStartBtn.textContent = 'RESUME';

                swStartBtn.classList.replace('bg-orange-500', 'bg-sky-500');

                swLapBtn.disabled = true;

            }

        }

 

        function resetStopwatch() {

            isRunning = false; clearInterval(swInterval); swElapsedTime = 0;

            swDisplay.textContent = '00:00.00';

            swStartBtn.textContent = 'START';

            swStartBtn.className = "flex-2 grow py-4 bg-sky-500 text-white rounded-2xl font-bold transition-all shadow-lg shadow-sky-500/20";

            swLapBtn.disabled = true; laps = [];

            lapsContainer.innerHTML = '<div class="text-slate-500 text-center text-sm mt-12">No laps recorded</div>';

        }

 

        function recordLap() { if (isRunning) { laps.unshift(swElapsedTime); renderLaps(); } }

        function renderLaps() {

            lapsContainer.innerHTML = laps.map((lap, index) => `

                <div class="flex justify-between py-3 border-b border-white/5 items-center">

                    <span class="text-slate-500 text-xs font-bold uppercase">Lap ${laps.length - index}</span>

                    <span class="font-mono text-white">${formatTime(lap)}</span>

                </div>

            `).join('');

        }

 

        // --- GEMINI API HELPERS ---

        async function callGemini(prompt, isJson = false) {

            const endpoint = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: prompt }] }]

            };

           

            if (isJson) {

                payload.generationConfig = {

                    responseMimeType: "application/json",

                    responseSchema: {

                        type: "OBJECT",

                        properties: {

                            items: {

                                type: "ARRAY",

                                items: {

                                    type: "OBJECT",

                                    properties: {

                                        time: { type: "STRING" },

                                        task: { type: "STRING" }

                                    }

                                }

                            }

                        }

                    }

                };

            }

 

            let retries = 0;

            const delay = (ms) => new Promise(res => setTimeout(res, ms));

           

            while (retries < 5) {

                try {

                    const response = await fetch(endpoint, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    const data = await response.json();

                    return data.candidates[0].content.parts[0].text;

                } catch (err) {

                    retries++;

                    await delay(Math.pow(2, retries) * 1000);

                }

            }

            throw new Error("Failed to reach Gemini after retries.");

        }

 

        // --- AI FEATURES ---

        async function generateAIQuote() {

            const quoteText = document.getElementById('quote-text');

            const original = quoteText.textContent;

            quoteText.textContent = "✨ Summoning wisdom...";

            quoteText.classList.add('shimmer');

 

            try {

                const now = new Date();

                const hour = now.getHours();

                const prompt = `Generate a short, powerful motivational quote (max 15 words) about time, productivity, or living in the moment. The current time is ${hour}:00, so make it relevant to that time of day (morning focus, afternoon energy, or evening reflection). No quotes marks.`;

                const quote = await callGemini(prompt);

                quoteText.textContent = quote.trim();

            } catch (e) {

                quoteText.textContent = original;

            } finally {

                quoteText.classList.remove('shimmer');

            }

        }

 

        async function generateSchedule() {

            const input = document.getElementById('ai-input').value;

            if (!input.trim()) return;

 

            const btn = document.getElementById('ai-plan-btn');

            const resultBox = document.getElementById('ai-result');

           

            btn.disabled = true;

            btn.innerHTML = '<div class="w-5 h-5 border-2 border-white border-t-transparent rounded-full animate-spin"></div>';

            resultBox.classList.remove('hidden');

            resultBox.innerHTML = '<div class="w-full h-32 shimmer rounded-xl"></div>';

 

            try {

                const prompt = `Convert the following disorganized schedule into a structured list of tasks with times. User Input: "${input}". Return a list of JSON objects with keys "time" and "task". Current time is ${new Date().toLocaleTimeString()}.`;

                const response = await callGemini(prompt, true);

                const data = JSON.parse(response);

 

                resultBox.innerHTML = `

                    <div class="space-y-3">

                        <div class="text-sky-400 text-[10px] uppercase font-bold tracking-widest mb-2 italic">✨ AI Optimized Schedule</div>

                        ${data.items.map(item => `

                            <div class="flex gap-4 items-start p-2 rounded-lg hover:bg-white/5 transition-colors">

                                <span class="text-sky-500 font-mono text-sm whitespace-nowrap">${item.time}</span>

                                <span class="text-slate-200 text-sm">${item.task}</span>

                            </div>

                        `).join('')}

                    </div>

                `;

            } catch (e) {

                resultBox.innerHTML = `<div class="text-red-400 text-sm p-4">Sorry, couldn't organize that right now. Please try again.</div>`;

            } finally {

                btn.disabled = false;

                btn.innerHTML = '<span>GENERATE ✨</span>';

            }

        }
